class IUser      
  def initialize
    raise NotImplementedError.new("#{self.class.name} is a interface.");
  end;
      
  def print
    raise NotImplementedError.new("#{self.class.name} is a interface.");
  end;
  
  public :initialize;
  public :print;
end;

class StudentUser < IUser      
  def initialize(n, a, s)
    @name = n;
    @age = a;
    @school = s;
  end;
      
  def print
    printf 'Name : ' + @name + ' Age : ' + @age.to_s + ' School : ' + @school;
  end;
  
  public :initialize;
  public :print;
end;
  
su = StudentUser.new('zhuzhen', 23, 'Number 9');
su.print;
