class User
{
   private $name;

   public function __construct($n)
   {
      $this->name = $n;
   }

   public function prints()
   {
      echo 'Name : ' . $this->name . '<br/>';
   }

   public function __destruct()
   {
      echo 'Free...';
   }
}

$u = new User('zhuzhen');
$u->prints();
